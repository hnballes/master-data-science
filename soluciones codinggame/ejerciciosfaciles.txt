import sys
import math

n = int(input()) 
q = int(input())  
diccionario={}

for i in range(n):
    ext, mt = input().split()
    ext1=ext.lower()
    diccionario[ext1]=mt
    
for i in range(q):
    fname = input()
    fname1=fname.lower()
    x=[idx for idx, x in enumerate(fname1) if x=='.']
    if x:
        if fname1[x[-1]+1:] in diccionario:
            print(diccionario[fname1[x[-1]+1:]])
        else:
            print("UNKNOWN")
    else:
        print("UNKNOWN")
